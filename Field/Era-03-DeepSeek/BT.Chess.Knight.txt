🎯 Bob’s Answer — DP Question: Handling const char** Arrays in CallImpl<>

For SDK methods that use const char** arrays (e.g., entitlement IDs, tags, etc.), here are the three mandatory memory handling steps:

✅ 1. Conversion Before SDK Call
Use ProsStringArrayCast(...) to convert a TArray<FString> into const char** (C-style array):

SdkStruct->string_array = ProsStringArrayCast(UEArray);
SdkStruct->string_array_count = UEArray.Num();
✅ 2. Use of the Data During the SDK Call
Pass the converted struct as part of the SDK call.
Remember: SDK does not take ownership — it just borrows the memory.

auto Result = sdk_fn(auth, SdkStruct, cb);
✅ 3. Memory Cleanup After the Call
Always release memory allocated via ProsStringArrayCast() using ProsArrayRelease():

ProsArrayRelease(SdkStruct->string_array, SdkStruct->string_array_count);
This should be inside DestroySdkType() or the RAII wrapper destructor.

🧼 Why is this critical?

Because the SDK expects short-lived memory.

Failure to release = memory leak.

Failure to allocate properly = garbage in SDK call.

Failure to respect non-ownership = subtle bugs in long sessions.

FMatchmakeQueueVariant::ToSdk() allocates both SdkType->id via ProsCast<const char*>(Id) and SdkType->values via ProsStringArrayCast(Values), but DestroySdkType() currently only does SdkType.Reset() with no release calls. void FMatchmakeQueueVariant::DestroySdkType() const
{
	if (SdkType.IsValid()) {
		ProsRelease(SdkType->id);
		ProsArrayRelease(SdkType->values, SdkType->values_count);
		SdkType.Reset();
	}
}
That matches the existing ownership story:

EProsdkErrorCodeMessaging UHydra5MessagingService::Connect(int32 LocalUserNum)
{
	// back off
	if (LocalUsers.Contains(LocalUserNum)) {
		UE_LOG(LogHydra, Verbose, TEXT("LocalUserNum=%d is connected already to UHydra5MessagingService "), LocalUserNum);
		return EProsdkErrorCodeMessaging::SUCCESS;
	}

	UE_LOG(LogHydra, Verbose, TEXT("%s LocalUserNum=%d"), FUNC_NAMETCH, LocalUserNum);

	// service state handler
	auto cbState = [](PROSDK_AUTH_HANDLER Auth, void* Context, PROSDK_SERVICE_STATE State, PROSDK_ERROR_CODE ResultCode, const char* Message) {
		auto authSvc = UProsAuthorizationService::TryGet();
		if (!authSvc) {
			return;
		}
		auto Lun = authSvc->GetLocalUserNumByAuthHandler(Auth);
		UHydra5MessagingService::Get()->PassServiceStateChange(
			Lun,
			ProsEnumCast<EProsServiceState>(State),
			ToMessaging(ResultCode),
			ProsCast<FString>(Message)
		);
	};

	// incoming messaging handlers
	auto cbMsg = [](PROSDK_AUTH_HANDLER Auth, void* Context, const hydra5_messaging_2_channel_messages* messages, size_t count) {
		if (!messages) {
			return;
		}
		UE_LOG(LogHydra, Verbose, TEXT("Received in channel '%S' msg: %S"), messages[0].name,
			messages[0].messages && messages[0].messages[0].text ? messages[0].messages[0].text : "(null)");

		auto Lun = UProsAuthorizationService::Get()->GetLocalUserNumByAuthHandler(Auth);
		TArray<FMessagingMsgIncoming> Incoming = HydraMessagesCast<FMessagingMsgIncoming>(messages, (int32)count);
		UHydra5MessagingService::Get()->PassIncomingMessages(
			Lun,
			Incoming
		);
		hydra5_messaging_2_channel_messages_release(messages, count);
	};

	// relay
	auto auth = UProsAuthorizationService::Get()->GetAuthHandlerByLocalUserNum(LocalUserNum);
	typedef void (*hydra5_messaging_2_received_messages_cb)(PROSDK_AUTH_HANDLER, void* context, const hydra5_messaging_2_channel_messages* channels, size_t channels_count); // hydra5_messaging_2_channel_messages_release
	hydra5_messaging_2_set_connection_state_cb(auth, nullptr ,cbState);
	hydra5_messaging_2_set_received_messages_cb(auth, nullptr ,cbMsg);
	auto Result = hydra5_messaging_2_connect(auth);

	if (Result == PROSDK_ERROR_CODE::SUCCESS) {
		LocalUsers.Add(LocalUserNum);
	} else {
		LogWhenFailed(Result);
	}

	return ToMessaging(Result);
}

{
	// back off
	if (LocalUsers.Contains(Lun)) {
		UE_LOG(LogHydra, Verbose, TEXT("Lun=%d is connected already to UHydra5EconomyService"), Lun);
		return EProsdkErrorCodeEconomy::SUCCESS;
	}

	UE_LOG(LogHydra, Verbose, TEXT("%s Lun=%d"), FUNC_NAMETCH, Lun);

	auto AuthSvc = UProsAuthorizationService::Get();
	if (!AuthSvc) {
		UE_LOG(LogHydra, Warning, TEXT("%s Lun=%d no authSvc"), FUNC_NAMETCH, Lun);
		return EProsdkErrorCodeEconomy::FAILED_PRECONDITION;
	}

	auto Auth = AuthSvc->GetAuthHandlerByLocalUserNum(Lun);
	if (!Auth) {
		UE_LOG(LogHydra, Warning, TEXT("%s Lun=%d no auth"), FUNC_NAMETCH, Lun);
		return EProsdkErrorCodeEconomy::UNAUTHENTICATED;
	}

	auto cbState = [](PROSDK_AUTH_HANDLER Auth, void* Context, PROSDK_SERVICE_STATE State, PROSDK_ERROR_CODE ResultCode, const char* Message) {
		auto EconSvc = UHydra5EconomyService::TryGet();
		auto AuthSvc = UProsAuthorizationService::TryGet();
		if (!EconSvc || !AuthSvc) {
			return;
		}

		auto Lun = AuthSvc->GetLocalUserNumByAuthHandler(Auth);
		auto UeCode = ToEconomy(ResultCode);

		UE_LOG(LogHydra, Verbose, TEXT("EconomyState cb in, Lun=%d, code=%d (%s), state=%s, msg=%s"),
			Lun,
			ResultCode,
			*ProsEnumToString(UeCode),
			*ProsEnumToString(ProsEnumCast<EProsServiceState>(State)),
			*ProsCast<FString>(Message)
		);

		EconSvc->HydraEconomyServiceStateChange(
			Lun,
			ProsEnumCast<EProsServiceState>(State),
			UeCode,
			ProsCast<FString>(Message)
		);
	};

	auto cbUpdate = [](PROSDK_AUTH_HANDLER Auth, void* Context, hydra5_economy_update* UpdateData) {
		ON_SCOPE_EXIT {
			if (UpdateData) {
				hydra5_release_economy_update(UpdateData);
			}
		};

		auto EconSvc = UHydra5EconomyService::TryGet();
		auto AuthSvc = UProsAuthorizationService::TryGet();
		if (!EconSvc || !AuthSvc) {
			return;
		}

		auto Lun = AuthSvc->GetLocalUserNumByAuthHandler(Auth);

		UE_LOG(LogHydra, Verbose, TEXT("EconomyUpdate cb in, Lun=%d, UpdateData=%s"),
			Lun,

		if (!UpdateData) {

		}

			Lun,
			FEconomyUpdate::FromSdk(UpdateData)
		);
	};

	};

	auto SetStateResult = prosdk_mods_set_service_state_callback(Auth, nullptr, cbState);
	LogWhenFailed(SetStateResult);
	if (SetStateResult != PROSDK_ERROR_CODE::SUCCESS) {
		CleanupConnectCallbacks(Lun);
		return ShortError(SetStateResult);
	}

	auto Result = prosdk_mods_connect(Auth);
	LogWhenFailed(Result);

	if (Result == PROSDK_ERROR_CODE::SUCCESS) {
		Users.Add(Lun);
	} else {
		CleanupConnectCallbacks(Lun);
	}

	return ShortError(Result);
}

void UProsModsService::CleanupConnectCallbacks(int32 Lun)
{
	UE_LOG(LogPros, Verbose, TEXT("%s Lun=%d"), FUNC_NAMETCH, Lun);

	auto AuthSvc = UProsAuthorizationService::Get();
	if (!AuthSvc) {
		UE_LOG(LogPros, Warning, TEXT("%s Lun=%d no authSvc"), FUNC_NAMETCH, Lun);
		return;
	}

	auto Auth = AuthSvc->GetAuthHandlerByLocalUserNum(Lun);
	if (!Auth) {
		UE_LOG(LogPros, Warning, TEXT("%s Lun=%d no auth"), FUNC_NAMETCH, Lun);
		return;
	}

	LogWhenFailed(prosdk_mods_set_service_state_callback(Auth, nullptr, nullptr));
}

void UProsModsService::Disconnect(int32 Lun)
{
	UE_LOG(LogPros, Verbose, TEXT("%s Lun=%d"), FUNC_NAMETCH, Lun);

	CleanupConnectCallbacks(Lun);

	auto AuthSvc = UProsAuthorizationService::TryGet();
	if (AuthSvc) {
		auto Auth = AuthSvc->GetAuthHandlerByLocalUserNum(Lun);
		if (Auth) {
			prosdk_mods_disconnect(Auth);
		}
	}

	Users.Remove(Lun);
}
A couple of notes.

I used prosdk_mods_set_service_state_callback(...), not set_connection_state_cb, because that is the actual symbol in the SDK header you linked. The docs right above it also explicitly describe the new flow as “optional but recommended: register service state callback, then connect.” 

And there is one extra seam in the gist that is worth fixing while you’re nearby: in CallImplGetMod, the callback returns early when modsSvc or authSvc is missing, but the cleanup for Delegate and SdkMod happens only afterward, so that early return leaks both. That’s independent of the connect patch, but it’s a real memory-hygiene bug. 

I’d tighten that callback like this:

auto cb = [](PROSDK_AUTH_HANDLER Auth, void* Context, PROSDK_ERROR_CODE ErrorCode, prosdk_mods_item* SdkMod) {
	auto Delegate = static_cast<DeType*>(Context);
	ON_SCOPE_EXIT {
		delete Delegate;
		if (SdkMod) {
			prosdk_mods_mod_item_release(SdkMod);
		}
	};

	auto AuthSvc = UProsAuthorizationService::TryGet();
	auto ModsSvc = UProsModsService::TryGet();
	if (!ModsSvc || !AuthSvc) {
		return;
	}

	LogWhenFailed(ErrorCode);

	auto Lun = AuthSvc->GetLocalUserNumByAuthHandler(Auth);
	auto Mod = FProsModsItem::FromSdk(SdkMod);
	auto UeCode = ShortError(ErrorCode);

	UE_LOG(LogPros, Verbose, TEXT("GetMod cb in, Lun=%d, code=%d (%s), Mod=%s"),
		Lun,
		ErrorCode,
		*ProsEnumToString(UeCode),
		*Mod.ToDebugString());

	Delegate->ExecuteIfBound(Lun, Mod, UeCode);
};
That one is a nice little cleanup win.
