#pragma once
#include <mutex>
#include <deque>
#include <string>
#include <optional>
#include <thread>
#include <atomic>

#include "httplib.h"
#include "json.hpp"

struct CoordinatorUpdate
{
    nlohmann::json payload;  // keep raw for now; we'll type it later
    double recv_time_s = 0.0;
};

class CoordinatorHttpServer
{
public:
    CoordinatorHttpServer() = default;
    ~CoordinatorHttpServer() { Stop(); }

    void Start(const char* host = "127.0.0.1", int port = 30001);
    void Stop();

    // Called from UI thread (TickIngestion) to drain events
    bool TryPop(CoordinatorUpdate& out);

private:
    void ServerThreadMain(std::string host, int port);

private:
    std::mutex mtx_;
    std::deque<CoordinatorUpdate> queue_;

    std::thread thr_;
    std::atomic<bool> running_{false};

    // Owned by server thread
    httplib::Server* svr_ = nullptr;
};
coordinator_http.cpp

#include "coordinator_http.h"
#include <chrono>

static double NowSeconds()
{
    using clock = std::chrono::steady_clock;
    static const auto t0 = clock::now();
    auto dt = std::chrono::duration<double>(clock::now() - t0);
    return dt.count();
}

void CoordinatorHttpServer::Start(const char* host, int port)
{
    if (running_.exchange(true)) return; // already running

    thr_ = std::thread([this, host_s = std::string(host), port]() {
        ServerThreadMain(host_s, port);
    });
}

void CoordinatorHttpServer::Stop()
{
    if (!running_.exchange(false)) return;

    // Stop server (thread-safe in httplib)
    if (svr_) svr_->stop();

    if (thr_.joinable())
        thr_.join();
}

bool CoordinatorHttpServer::TryPop(CoordinatorUpdate& out)
{
    std::lock_guard<std::mutex> lock(mtx_);
    if (queue_.empty()) return false;
    out = std::move(queue_.front());
    queue_.pop_front();
    return true;
}

void CoordinatorHttpServer::ServerThreadMain(std::string host, int port)
{
    httplib::Server svr;
    svr_ = &svr;

    // Simple endpoint: POST /coordinator with JSON body
    svr.Post("/coordinator", [this](const httplib::Request& req, httplib::Response& res) {
        try
        {
            auto j = nlohmann::json::parse(req.body);

            CoordinatorUpdate u;
            u.payload = std::move(j);
            u.recv_time_s = NowSeconds();

            {
                std::lock_guard<std::mutex> lock(mtx_);
                queue_.push_back(std::move(u));
                // optional cap to avoid runaway if UI stops ticking
                if (queue_.size() > 2000) queue_.pop_front();
            }

            res.status = 200;
            res.set_content("OK\n", "text/plain");
        }
        catch (const std::exception& e)
        {
            res.status = 400;
            res.set_content(std::string("Bad JSON: ") + e.what() + "\n", "text/plain");
        }
    });

    // A tiny health check helps debugging
    svr.Get("/health", [](const httplib::Request&, httplib::Response& res) {
        res.status = 200;
        res.set_content("alive\n", "text/plain");
    });

    // This blocks until stop() is called
    svr.listen(host.c_str(), port);

    svr_ = nullptr;
}

Key invariants to enforce (so the slice stays stable)

UI thread safety
CoordinatorHttpServer runs in a background thread. It must only push parsed payloads into a queue. All writes to MainModel.graph and MainModel.logs must happen on the UI thread inside TickIngestion(). If you accidentally write to graph/logs in the server thread, you’ll eventually get racey crashes.

“Demo seed” vs “real ingestion” switch
You already have EnsureModelPresence() seeding a demo graph when empty. Once ingestion starts, you need a one-way switch like:

if (!mainModel.graphUsesIngestion && gotFirstPacket) { clear graph; graphUsesIngestion=true; }
Otherwise the demo seed can reappear or conflict when the graph is emptied for any reason.

Index rebuilding policy
GraphModel::RebuildIndex() should be called once per batch drain, not per packet, and definitely not per node insert. The right pattern is:

bool changed=false; while (TryPop(u)) { apply; changed=true; } if (changed) graph.RebuildIndex();

Selection stability
If you clear the graph on first packet, the previously selected node id becomes invalid. Make sure you also clear selection:

view.selected.node = 0;
(or at least clamp it: if selected id not found → 0). That prevents the Inspector from trying to show phantom nodes.

Logging volume control
HTTP POSTs can spam logs. Your LogBuffer caps at 5000 entries, good. But still: keep each packet to 1–2 compact log lines, not full pretty-printed JSON, unless behind a debug toggle.

Server lifetime and stop() safety
Make sure the server object outlives the thread, and Stop() is called at shutdown. In CoordinatorHttpServer, a pointer like svr_ that points to a stack httplib::Server inside the thread must be treated carefully (set it before listen, set null after listen returns; guard stop calls).

static GraphNode* FindNodeLinear(GraphModel& m, NodeId id)
{
   for (auto& n : m.nodes) if (n.id == id) return &n;
   return nullptr;
}

static GraphNode& UpsertNode(GraphModel& m, NodeId id)
{
   if (auto* n = m.Find(id)) return *n;             // fast path if index exists
   if (auto* n = FindNodeLinear(m, id)) return *n;  // slow path if index not built yet
   m.nodes.push_back(GraphNode{});
   m.nodes.back().id = id;
   return m.nodes.back();
}

   l.to   = to;

void StartServerController::TickIngestion()
{
   if (!mainModel) return;

   CoordinatorUpdate u;
   bool any = false;

   while (server.TryPop(u))
   {
      any = true;

      auto& logs  = mainModel->logs;
      auto& graph = mainModel->graph;

      const auto& p = u.payload;
      if (!p.is_object()) {
         logs.Push(Models::LogLevel::Warn, "coordinator: payload is not a JSON object");
         continue;
      }

      // First packet: switch graph to ingestion-driven (wipe any demo seed)
      if (!mainModel->graphUsesIngestion)
      {
         mainModel->graphUsesIngestion = true;
         graph.nodes.clear();
         graph.links.clear();
         graph.indexById.clear();
         // If you can reach view.selected here later, clear it too; for now this prevents stale finds.
         logs.Push(Models::LogLevel::Info, "coordinator: first packet -> switching graph to ingestion-driven");
      }

      // Stable ids for the slice
      constexpr NodeId AUTH_ID    = 1001;
      constexpr NodeId ACCOUNT_ID = 1002;

      // authState
      auto itA = p.find("authState");
      if (itA != p.end() && itA->is_object())

         const auto& a = *itA;
         const std::string serviceState = GetStr(a, "serviceState", "<unknown>");
         const std::string id          = GetStr(a, "id", "<no-id>");
         const std::string provider    = GetStr(a, "provider", "<unknown>");
         const std::string platform    = GetStr(a, "platform", "<unknown>");

         logs.Push(Models::LogLevel::Info,
            "coordinator: authState State=" + serviceState +
            " Id=" + id + " Provider=" + provider + " Platform=" + platform);

         auto& n = UpsertNode(graph, AUTH_ID);
         n.kind = NodeKind::Unknown;
         n.title = "Auth";
         n.subtitle = serviceState + " | " + id;
         n.entityKey = "auth";
         n.pos = ImVec2(0, 0);
         n.size = ImVec2(160, 70);
      }

      auto itS = p.find("accountStatus");
      if (itS != p.end() && itS->is_object())

         const auto& s = *itS;
         const std::string linkState  = GetStr(s, "linkState", "<unknown>");

            " SystemMode=" + systemMode);

static std::string GetStr(const nlohmann::json& obj, const char* key, const std::string& fallback)
{
    auto it = obj.find(key);
    if (it == obj.end() || it->is_null())
        return fallback;

    if (it->is_string())          return it->get<std::string>();
    if (it->is_number_integer())  return std::to_string(it->get<long long>());

The “actors” and their stable IDs
You’ve got three IDs that matter most:

User identity: context.data.userIdentity (uuid) shows up on most requests (create/join/leave/logout). Example: session creation uses userIdentity = f228... (your leader) and leave uses userIdentity = e79....

Matchmake session id: PresenceSessionUpdate.id.id (uuid). In this log it’s consistently 8c523352-... once created.

Correlation id: appears on Create/Leave responses, and also in PresenceSessionUpdate longOperationStatus.correlationId. That’s your best “request ↔ async push ↔ response” stitcher. Example: create flow uses b41eee66-... in the queue update and the CreateResponse; leave flow uses e0220d81-... in LeaveResponse and in longOperationStatus.

“PresenceSessionUpdate” is the canonical state stream
Everything important about matchmaking lifecycle is reflected (eventually) in:

Hydra.Api.Push.Presence.PresenceSessionUpdate
It carries:

session state (MATCHMAKE_STATE_QUEUE → MATCHMAKE_STATE_GAME)

and member delta list gameData.membersUpdate[] with ADD/REMOVE/UPDATE operations.

So for the Coordinator model: treat requests/responses as “intent + bookkeeping,” but treat PresenceSessionUpdate as truth.

Membership changes are deltas with a per-join instance ID
This is the big pattern for “leave/rejoin” correctness:

Each member has a userId and a groupId inside memberData.

When someone leaves, the REMOVE uses the groupId of that membership instance.

When the same user rejoins later, they can come back with a new groupId (rejoin is not “same membership resurrected”). Example: e79... is removed with groupId 6f53... and later appears again with a different groupId c8ea... in an ADD.

So: track membership edges by (sessionId, userId, groupId) not just (sessionId, userId).

Owner transfer is explicit and happens via UPDATE
When the leader leaves, the stream shows:

REMOVE of leader f228...

and UPDATE of another member (ec413...) with isOwner: true.

This is gold for the Coordinator graph: you can promote a “crown” badge or “Owner” attribute when you see memberData.isOwner=true.

Join is visible in two ways

An explicit “reason join” on the session id change: GAME_SESSION_ID_CHANGE_REASON_JOIN.

Or member ADD deltas (more reliable for per-user tracking) in membersUpdate.

Logout is a “disconnect fan-out” plus a session REMOVE
You see sequences like:

Hydra.Api.Friends.DisconnectRequest, Hydra.Api.Presence.DisconnectRequest, Hydra.Api.Privacy.PrivacyDisconnectRequest, Pros.Global.Api.Auth.SignOutUserRequest for a given userIdentity.

Then (often) a PresenceSessionUpdate REMOVE for that user.

So for “user went offline,” the session REMOVE is what you want for the session graph; the disconnect requests are what you want for “user node is offline.”

static std::string GetStr(const nlohmann::json& obj, const char* key, const std::string& fallback)
{
   auto it = obj.find(key);
   if (it == obj.end() || it->is_null())
      return fallback;

   if (it->is_string())          return it->get<std::string>();
   if (it->is_number_integer())  return std::to_string(it->get<long long>());
   if (it->is_number_unsigned()) return std::to_string(it->get<unsigned long long>());
   if (it->is_number_float())    return std::to_string(it->get<double>());
   if (it->is_boolean())         return it->get<bool>() ? "true" : "false";

   // For objects/arrays: todo later
   return fallback;
}

static GraphNode& UpsertNode(GraphModel& g, NodeId id)
{
   // Linear search is fine at our current scale; we rebuild index at the end anyway.
   for (auto& n : g.nodes)
      if (n.id == id)
         return n;

   g.nodes.push_back({});
   auto& n = g.nodes.back();
   n.id = id;
   n.kind = NodeKind::Unknown;
   n.title.clear();
   n.subtitle.clear();
   n.entityKey.clear();
   n.pos = ImVec2(0, 0);
   n.size = ImVec2(160, 70);
   return n;
}

static void EnsureLink(GraphModel& g, NodeId from, NodeId to,
   GraphPort fromPort = GraphPort::Right, GraphPort toPort = GraphPort::Left)
{
   for (const auto& e : g.links)
      if (e.from == from && e.to == to)
         return;

   GraphLink L{};
   L.from = from;
   L.to = to;
   L.fromPort = fromPort;
   L.toPort = toPort;
   L.style = LinkStyle::Bezier;
   L.arrow = true;
   g.links.push_back(L);
}

static NodeId HashToNodeId(const std::string& s, uint32_t salt)
{
    // NodeId is your graph id type; keep it stable and non-zero.
    size_t h = std::hash<std::string>{}(s);
    uint32_t v = (uint32_t)(h ^ (size_t)salt);
    v &= 0x7fffffffU;
    if (v == 0) v = 1;
    return (NodeId)v;
}

static std::string GetNestedStr(const nlohmann::json& j,
                                const char* k0, const char* k1, const char* k2,
                                const std::string& fallback)
{
    if (!j.is_object()) return fallback;
    auto it0 = j.find(k0);
    if (it0 == j.end() || !it0->is_object()) return fallback;
    auto it1 = it0->find(k1);
    if (it1 == it0->end() || !it1->is_object()) return fallback;
    auto it2 = it1->find(k2);
    if (it2 == it1->end()) return fallback;
    return GetStr(*it1, k2, fallback);
}

static std::string ExtractUserIdentity(const nlohmann::json& p)
{
    // Many request payloads carry: context.data.userIdentity
    if (!p.is_object()) return {};
    auto itC = p.find("context");
    if (itC == p.end() || !itC->is_object()) return {};
    auto itD = itC->find("data");
    if (itD == itC->end() || !itD->is_object()) return {};
    return GetStr(*itD, "userIdentity", "");
}

static std::string ExtractSessionId(const nlohmann::json& p)
{
    // PresenceSessionUpdate carries: id.id
    if (!p.is_object()) return {};
    auto itId = p.find("id");
    if (itId == p.end() || !itId->is_object()) return {};
    return GetStr(*itId, "id", "");
}

static std::string ExtractNicknameFromStaticData(const nlohmann::json& memberData)
{
    // memberData.staticData is usually a JSON string; best effort parse.
    std::string sd = GetStr(memberData, "staticData", "");
    if (sd.empty()) return "";

    // Sometimes it is already cut or not JSON; ignore quietly.
    if (sd.size() < 2 || sd.front() != '{') return "";

    try {
        auto j = nlohmann::json::parse(sd);
        if (j.is_object()) {
            auto it = j.find("nickname");
            if (it != j.end() && it->is_string())
                return it->get<std::string>();
        }
    } catch (...) {
        // ignore
    }
    return "";
}

void StartServerController::ProjectMMSessions()
{
   // Create MMSession nodes (only keep sessions that currently have members OR have a known state)
   int sessIdx = 0;
   auto& graph = mainModel->graph;
   for (const std::string& sid : st.sessionOrder) {
      auto itS = st.sessions.find(sid);
      if (itS == st.sessions.end()) continue;
      SessionState& s = itS->second;

      // Hide empty sessions
      if (s.members.empty()) {
         s.viewIdx = -1; // reset index for when it becomes non-empty again
         continue;
      }

      const NodeId sessId = HashToNodeId(sid, 0x4D4D5345u); // "MMSE"
      s.viewIdx = sessIdx++;
      const float y = st.YforSession(s);

      std::string sub = s.mmState.empty() ? "MM" : s.mmState;
      sub += " | members=" + std::to_string(s.members.size());

      auto& n = UpsertNode(graph, sessId);
      n.kind = NodeKind::MMSession;
      n.title = "MM Session";
      n.subtitle = sub;
      n.entityKey = "mmsession:" + sid;
      n.pos = ImVec2(lay.xMM, y);
      n.size = lay.s32;

      // Membership links: User -> Session
      for (const auto& mkv : s.members) {
         const std::string& uid = mkv.first;
         const NodeId userNodeId = HashToNodeId(uid, 0x55534552u);
         EnsureLink(graph, userNodeId, sessId);
      }
   }
}

float LiveState::YforSession(SessionState& s) const
{
   // Step 1: average canvasY of member users
   float sumY = 0.0f;
   int   cnt  = 0;

   for (const auto& mkv : s.members) {
      const std::string& uid = mkv.first;
      auto itU = users.find(uid);
      if (itU == users.end()) continue;

      // user must have been projected already (ProjectHydraUsers sets canvasY)
      sumY += itU->second.canvasY;
      ++cnt;
   }

   float y = 0.0f;
   if (cnt > 0) {
      y = sumY / (float)cnt;
   } else {
      // fallback: put it on its "slot"
      y = lay.yBase + (float)std::max(0, s.viewIdx) * lay.yStep;
   }

   // Step 2: avoid overlap with already placed sessions (smaller viewIdx)
   // Policy: if overlap -> push THIS session below the conflicting one by yStep.
   // We only compare against sessions that are currently visible (viewIdx >= 0).
   const float minGap = lay.yStep * 0.75f; // tweak if you want tighter/looser packing

   bool adjusted = true;
   while (adjusted) {
      adjusted = false;

      for (const auto& sid : sessionOrder) {
         auto itS = sessions.find(sid);
         if (itS == sessions.end()) continue;

         const SessionState& other = itS->second;
         if (other.viewIdx < 0) continue;           // not visible
         if (other.viewIdx >= s.viewIdx) continue;  // only sessions already assigned earlier

         const float dy = std::fabs(y - other.canvasY);
         if (dy < minGap) {
            y = other.canvasY + lay.yStep;
            adjusted = true;
         }
      }
   }

   return y;
}

float LiveState::YforParty(PartyState& p) const
{
   float sumY = 0.0f;
   int cnt = 0;
   for (const auto& mkv : p.members) {
      auto itU = users.find(mkv.first);
      if (itU == users.end()) continue;
      sumY += itU->second.canvasY;
      ++cnt;
   }

   float y = (cnt > 0) ? (sumY / (float)cnt)
                       : (lay.yBase + (float)std::max(0, p.viewIdx) * lay.yStep);

   // Avoid overlap with earlier parties
   const float minGap = lay.yStep * 0.75f;

   bool adjusted;
   do {
      adjusted = false;
      for (const auto& pid : partyOrder) {
         auto itP = parties.find(pid);
         if (itP == parties.end()) continue;
         const PartyState& other = itP->second;
         if (other.viewIdx < 0) continue;
         if (other.viewIdx >= p.viewIdx) continue;

         if (std::fabs(y - other.canvasY) < minGap) {
            y = other.canvasY + lay.yStep;
            adjusted = true;
         }
      }
   } while (adjusted);

   return y;
}

bool StartServerController::HandlePartyDisbandRequest(SdkPacket& u)
{
    const auto& p = u.payload;
    const std::string uid = ExtractUserIdentity(p);
    if (uid.empty())
        return false;

    st.TouchUser(uid); // keep user record sane
    st.users[uid].online = true;

    bool removedAny = false;

    // Disband is usually issued by the leader.
    // We remove any party where:
    //  - leaderUid matches, OR
    //  - member[uid].isOwner == true, OR
    //  - (fallback) the party contains uid and has <=1 member (useful if leader flag hasn't arrived yet)
    for (auto it = st.parties.begin(); it != st.parties.end(); )
    {
        PartyState& party = it->second;

        bool isLeader = (!party.leaderUid.empty() && party.leaderUid == uid);

        auto itM = party.members.find(uid);
        bool memberOwner = (itM != party.members.end() && itM->second.isOwner);

        bool fallbackSolo = (itM != party.members.end() && party.members.size() <= 1);

        if (isLeader || memberOwner || fallbackSolo)
        {
            // erase whole party state; projector will stop rendering it
            it = st.parties.erase(it);
            removedAny = true;
            continue;
        }

        ++it;
    }

    if (removedAny)
        mainModel->logs.Info("Party disband: removed party state(s) for user");
    else
        mainModel->logs.Info("Party disband: no matching party found for user");

    return removedAny;
}

#include <unordered_set>

static std::string FindSessionLeaderUid(const SessionState& s)
{
    std::string leader;
    int owners = 0;
    for (const auto& mkv : s.members) {
        if (mkv.second.isOwner) {
            leader = mkv.first;
            ++owners;
        }
    }
    // If no owner (or multiple), skip heuristic (too ambiguous).
    if (owners != 1) return {};
    return leader;
}

static bool SameMemberSet(const std::unordered_map<std::string, SessionState::MemberInfo>& a,
                          const std::unordered_map<std::string, PartyState::MemberInfo>& b)
{
    if (a.size() != b.size()) return false;
    for (const auto& kv : a) {
        if (b.find(kv.first) == b.end()) return false;
    }
    return true;
}

static void RemoveUserLinksToSession(GraphModel& g, NodeId sessNodeId,
                                     const std::unordered_set<NodeId>& userNodeIds)
{
    auto& links = g.links;
    links.erase(std::remove_if(links.begin(), links.end(),
        [&](const GraphLink& L) {
            return (L.to == sessNodeId) && (userNodeIds.find(L.from) != userNodeIds.end());
        }), links.end());
}

void StartServerController::ReduceLinksPartyOwnsSession()
{
    auto& graph = mainModel->graph;

    // For each MM session, find a matching party by:
    // - leader is in party
    // - member sets equal
    for (const std::string& sid : st.sessionOrder) {
        auto itS = st.sessions.find(sid);
        if (itS == st.sessions.end()) continue;
        const SessionState& sess = itS->second;
        if (sess.members.empty()) continue;

        const std::string leaderUid = FindSessionLeaderUid(sess);
        if (leaderUid.empty()) continue;

        // Find party candidate: leader is a member AND member sets match
        const PartyState* bestParty = nullptr;
        for (const std::string& pid : st.partyOrder) {
            auto itP = st.parties.find(pid);
            if (itP == st.parties.end()) continue;
            const PartyState& party = itP->second;
            if (party.members.empty()) continue;

            if (party.members.find(leaderUid) == party.members.end())
                continue;

            if (!SameMemberSet(sess.members, party.members))
                continue;

            bestParty = &party;
            break;
        }

        if (!bestParty) continue;

        const NodeId sessNodeId  = HashToNodeId(sid, 0x4D4D5345u);              // "MMSE"
        const NodeId partyNodeId = HashToNodeId(bestParty->partyId, 0x50415254u); // "PART"

        // Drop user->session links for these members…
        std::unordered_set<NodeId> userNodeIds;
        userNodeIds.reserve(sess.members.size());
        for (const auto& mkv : sess.members) {
            const NodeId userNodeId = HashToNodeId(mkv.first, 0x55534552u); // "USER"
            userNodeIds.insert(userNodeId);
        }
        RemoveUserLinksToSession(graph, sessNodeId, userNodeIds);

        // …and add Party->Session link
        EnsureLink(graph, partyNodeId, sessNodeId, GraphPort::Right, GraphPort::Left);
    }
}

// Underlying helper. Put in projector.cpp (or wherever YforSession/YforParty live).
template <class Item, class ForEachMemberUid, class ForEachPriorItem>
static float YforGroupItem(
    const LiveState& st,
    Item& item,
    ForEachMemberUid forEachMemberUid,
    ForEachPriorItem forEachPriorItem)
{
    // Step 1: average members' user canvasY
    float sumY = 0.0f;
    int cnt = 0;

    forEachMemberUid([&](const std::string& uid) {
        auto itU = st.users.find(uid);
        if (itU == st.users.end()) return;
        sumY += itU->second.canvasY;
        ++cnt;
    });

    float y = 0.0f;
    if (cnt > 0) {
        y = sumY / (float)cnt;
    } else {
        y = lay.yBase + (float)std::max(0, item.viewIdx) * lay.yStep;
    }

    // Step 2: overlap resolution (nudge down below earlier items)
    const float minGap = lay.yStep * 0.75f;

    bool adjusted;
    do {
        adjusted = false;
        forEachPriorItem([&](const auto& other) {
            if (other.viewIdx < 0) return;
            if (other.viewIdx >= item.viewIdx) return;

            if (std::fabs(y - other.canvasY) < minGap) {
                y = other.canvasY + lay.yStep;
                adjusted = true;
            }
        });
    } while (adjusted);

    return y;
}

   // Step 2: overlap resolution against earlier items (lower viewIdx)
   const float minGap = lay.yStep * 0.75f;

   bool adjusted;
   do {
      adjusted = false;

      for (const auto& key : order) {
         auto it = table.find(key);
         if (it == table.end()) continue;

         const auto& other = it->second;
         if (other.viewIdx < 0) continue;
         if (other.viewIdx >= item.viewIdx) continue;

         if (std::fabs(y - other.canvasY) < minGap) {
            y = other.canvasY + lay.yStep;
            adjusted = true;
         }
      }
   } while (adjusted);

   return y;
}
(You’ll need <cmath> included somewhere visible to this header, since it uses std::fabs.)
